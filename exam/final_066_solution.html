Student.java

public class Student extends Member {
  private int year;
  private double[] grades;
  
  public Student(String name, String id, String department) {
    super(name, id, department);
    grades = new double[4];
    year = 1;
  }
  
  public void setGrade(int year, double grade) {
    grades[year-1] = grade;
    if (year > this.year) {
      this.year = year;
    }
  }
  
  public double getGradeAverage() {
    double gradeSum = 0;
    for (int i = 0 ; i < year ; i++) {
      gradeSum += grades[i];
    }
    return gradeSum / year;
  }
  
  @Override
  public double getAverage() {
    return getGradeAverage();
  }

  public String toString() {
    return getInfo() + "\t" + getGradeAverage();
  }
}

Professor.java

public class Professor extends Member {
  private double[] lectures;
  
  public Professor(String name, String id, String department) {
    super(name, id, department);
    lectures = new double[5];
  }
  
  public void setLecture(int index, int lecture) {
    lectures[index] = lecture;
  }
  
  public double getLectureAverage() {
    int lectureSum = 0;
    for (int i = 0 ; i < lectures.length ; i++) {
      lectureSum += lectures[i];
    }
    return lectureSum / 5.0;
  }
  
  @Override
  public double getAverage() {
    return getLectureAverage();
  }

  public String toString() {
    return getInfo() + "\t" + getLectureAverage();
  }
}

Member.java

public abstract class Member {
  private String name;
  private String id;
  private String department;
  
  public Member(String name, String id, String department) {
    this.name = name;
    this.id = id;
    this.department = department;
  }
  
  public abstract double getAverage();
  public String getInfo() {
    if (this instanceof Professor) {
      return name + "\t" + id + "\t\t" + department + "\t"; 
    } else {
      return name + "\t" + id + "\t" + department + "\t";
    }
  }
}
